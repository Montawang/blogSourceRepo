### 顺便放个代码

```python
from random import uniform
def defCar():
    i = int(uniform(1, 4))
    return i

def cpt_posb():
    N = 1000000
    change = 0
    dischange = 0
    for i in range(N):
        car = defCar()
        doors = [1, 2, 3]
        choice = int(uniform(1, 4))
        if choice == car:
            dischange += 1
        else:
            change += 1
    print(change / N * 100, "%")
    print(dischange / N * 100, "%")

if __name__ == "__main__":
    cpt_posb()
```
